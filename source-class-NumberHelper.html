 98:         $places = 0;

100:             $places = $options;

105:         $before = $after = null;
106:         if (is_string($options) && !in_array($options, $separators)) {
107:             $before = $options;
108:         }
109:         $thousands = ',';
110:         if (!is_array($options) && in_array($options, $separators)) {

114:         if (!is_array($options) && in_array($options, $separators)) {

131: /**
132:  * Formats a number into a currency format.
133:  *
134:  * @param float $number
135:  * @param string $currency Shortcut to default options. Valid values are 'USD', 'EUR', 'GBP', otherwise set at least 'before' and 'after' options.

139:     function currency($number, $currency = 'USD', $options = array()) {
