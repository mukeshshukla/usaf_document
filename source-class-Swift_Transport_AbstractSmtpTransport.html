 55:   /**
 56:    * Creates a new EsmtpTransport using the given I/O buffer.

 58:    * @param Swift_Transport_IoBuffer $buf

 65:     $this->_buffer = $buf;

 70:    * Set the name of the local domain which Swift will identify itself as.
 71:    * This should be a fully-qualified domain name and should be truly the domain
 72:    * you're using.  If your server doesn't have a domain name, use the IP in square
 73:    * brackets (i.e. [127.0.0.1]).
 74:    * 
 75:    * @param string $domain
 76:    */

 79:     $this->_domain = $domain;

 84:    * Get the name of the domain Swift will identify as.

 93:   /**
 94:    * Start the SMTP connection.
 95:    */
 96:   public function start()
 97:   {
 98:     if (!$this->_started)
 99:     {
100:       if ($evt = $this->_eventDispatcher->createTransportChangeEvent($this))
101:       {
102:         $this->_eventDispatcher->dispatchEvent($evt, 'beforeTransportStarted');
103:         if ($evt->bubbleCancelled())
104:         {
105:           return;
106:         }
107:       }
108:       
109:       try
110:       {
111:         $this->_buffer->initialize($this->_getBufferParams());
112:       }
113:       catch (Swift_TransportException $e)
114:       {
115:         $this->_throwException($e);
116:       }
117:       $this->_readGreeting();
118:       $this->_doHeloCommand();
119:       
120:       if ($evt)
121:       {
122:         $this->_eventDispatcher->dispatchEvent($evt, 'transportStarted');
123:       }
124:       
125:       $this->_started = true;
126:     }
127:   }
128:   
129:   /**
130:    * Test if an SMTP connection has been established.
131:    * 
132:    * @return boolean
133:    */
134:   public function isStarted()
135:   {
136:     return $this->_started;
137:   }
138:   
139:   /**
140:    * Send the given Message.
141:    * 
142:    * Recipient/sender data will be retreived from the Message API.
143:    * The return value is the number of recipients who were accepted for delivery.
144:    * 
145:    * @param Swift_Mime_Message $message
146:    * @param string[] &$failedRecipients to collect failures by-reference
147:    * @return int
148:    */
149:   public function send(Swift_Mime_Message $message, &$failedRecipients = null)
150:   {
151:     $sent = 0;
152:     $failedRecipients = (array) $failedRecipients;
153:     
154:     if (!$reversePath = $this->_getReversePath($message))
155:     {
156:       throw new Swift_TransportException(
157:         'Cannot send message without a sender address'
158:         );
159:     }
160:     
161:     if ($evt = $this->_eventDispatcher->createSendEvent($this, $message))
162:     {
163:       $this->_eventDispatcher->dispatchEvent($evt, 'beforeSendPerformed');
164:       if ($evt->bubbleCancelled())
165:       {
166:         return 0;
167:       }
168:     }
169:     
170:     $to = (array) $message->getTo();
171:     $cc = (array) $message->getCc();
172:     $bcc = (array) $message->getBcc();
173:     
174:     $message->setBcc(array());
175:     
176:     try
177:     {
178:       $sent += $this->_sendTo($message, $reversePath, $to, $failedRecipients);
179:       $sent += $this->_sendCc($message, $reversePath, $cc, $failedRecipients);
180:       $sent += $this->_sendBcc($message, $reversePath, $bcc, $failedRecipients);
181:     }
182:     catch (Exception $e)
183:     {
184:       $message->setBcc($bcc);
185:       throw $e;
186:     }
187:     
188:     $message->setBcc($bcc);
189:     
190:     if ($evt)
191:     {
192:       if ($sent == count($to) + count($cc) + count($bcc))
193:       {
194:         $evt->setResult(Swift_Events_SendEvent::RESULT_SUCCESS);
195:       }
196:       elseif ($sent > 0)
197:       {
198:         $evt->setResult(Swift_Events_SendEvent::RESULT_TENTATIVE);
199:       }
200:       else
201:       {
202:         $evt->setResult(Swift_Events_SendEvent::RESULT_FAILED);
203:       }
204:       $evt->setFailedRecipients($failedRecipients);
205:       $this->_eventDispatcher->dispatchEvent($evt, 'sendPerformed');
206:     }
207:     
208:     $message->generateId(); //Make sure a new Message ID is used
209:     
210:     return $sent;
211:   }

213:   /**
214:    * Stop the SMTP connection.
215:    */
216:   public function stop()
217:   {
218:     if ($this->_started)
219:     {
220:       if ($evt = $this->_eventDispatcher->createTransportChangeEvent($this))
221:       {
222:         $this->_eventDispatcher->dispatchEvent($evt, 'beforeTransportStopped');
223:         if ($evt->bubbleCancelled())
224:         {
225:           return;
226:         }
227:       }
228:       
229:       try
230:       {
231:         $this->executeCommand("QUIT\r\n", array(221));
232:       }
233:       catch (Swift_TransportException $e) {}
234:       
235:       try
236:       {
237:         $this->_buffer->terminate();
238:       
239:         if ($evt)
240:         {
241:           $this->_eventDispatcher->dispatchEvent($evt, 'transportStopped');
242:         }
243:       }
244:       catch (Swift_TransportException $e)
245:       {
246:         $this->_throwException($e);
247:       }
248:     }
249:     $this->_started = false;
250:   }

280:   /**
281:    * Run a command against the buffer, expecting the given response codes.
282:    * 
283:    * If no response codes are given, the response will not be validated.
284:    * If codes are given, an exception will be thrown on an invalid response.
285:    * 
286:    * @param string $command
287:    * @param int[] $codes
288:    * @param string[] &$failures
289:    * @return string
290:    */
291:   public function executeCommand($command, $codes = array(), &$failures = null)
292:   {
293:     $failures = (array) $failures;
294:     $seq = $this->_buffer->write($command);
295:     $response = $this->_getFullResponse($seq);
296:     if ($evt = $this->_eventDispatcher->createCommandEvent($this, $command, $codes))
297:     {
298:       $this->_eventDispatcher->dispatchEvent($evt, 'commandSent');
299:     }
300:     $this->_assertResponseCode($response, $codes);
301:     return $response;
302:   }

342:   /** Stream the contents of the message over the buffer */
343:   protected function _streamMessage(Swift_Mime_Message $message)
344:   {
345:     $this->_buffer->setWriteTranslations(array("\r\n." => "\r\n.."));
346:     try
347:     {
348:       $message->toByteStream($this->_buffer);
349:       $this->_buffer->flushBuffers();
350:     }
351:     catch (Swift_TransportException $e)
352:     {
353:       $this->_throwException($e);
354:     }
355:     $this->_buffer->setWriteTranslations(array());
356:     $this->executeCommand("\r\n.\r\n", array(250));
357:   }

359:   /** Determine the best-use reverse path for this message */
360:   protected function _getReversePath(Swift_Mime_Message $message)
361:   {
362:     $return = $message->getReturnPath();
363:     $sender = $message->getSender();
364:     $from = $message->getFrom();
365:     $path = null;
366:     if (!empty($return))
367:     {
368:       $path = $return;
369:     }
370:     elseif (!empty($sender))
371:     {
372:       // Don't use array_keys
373:       reset($sender); // Reset Pointer to first pos
374:       $path = key($sender); // Get key
375:     }
376:     elseif (!empty($from))
377:     {
378:       reset($from); // Reset Pointer to first pos
379:       $path = key($from); // Get key
380:     }
381:     return $path;
382:   }

384:   /** Throw a TransportException, first sending it to any listeners */
385:   protected function _throwException(Swift_TransportException $e)
386:   {
387:     if ($evt = $this->_eventDispatcher->createTransportExceptionEvent($this, $e))
388:     {
389:       $this->_eventDispatcher->dispatchEvent($evt, 'exceptionThrown');
390:       if (!$evt->bubbleCancelled())
391:       {
392:         throw $e;
393:       }
394:     }
395:     else
396:     {
397:       throw $e;
398:     }
399:   }

401:   /** Throws an Exception if a response code is incorrect */
402:   protected function _assertResponseCode($response, $wanted)
403:   {
404:     list($code, $separator, $text) = sscanf($response, '%3d%[ -]%s');
405:     $valid = (empty($wanted) || in_array($code, $wanted));
406:     
407:     if ($evt = $this->_eventDispatcher->createResponseEvent($this, $response,
408:       $valid))
409:     {
410:       $this->_eventDispatcher->dispatchEvent($evt, 'responseReceived');
411:     }
412:     
413:     if (!$valid)
414:     {
415:       $this->_throwException(
416:         new Swift_TransportException(
417:           'Expected response code ' . implode('/', $wanted) . ' but got code ' .
418:           '"' . $code . '", with message "' . $response . '"'
419:           )
420:         );
421:     }
422:   }

424:   /** Get an entire multi-line response using its sequence number */
425:   protected function _getFullResponse($seq)
426:   {
427:     $response = '';
428:     try
429:     {
430:       do
431:       {
432:         $line = $this->_buffer->readLine($seq);
433:         $response .= $line;
434:       }
435:       while (null !== $line && false !== $line && ' ' != $line{3});
436:     }
437:     catch (Swift_TransportException $e)
438:     {
439:       $this->_throwException($e);
440:     }
441:     return $response;
442:   }

446:   /** Send an email to the given recipients from the given reverse path */
447:   private function _doMailTransaction($message, $reversePath,
448:     array $recipients, array &$failedRecipients)
449:   {
450:     $sent = 0;
451:     $this->_doMailFromCommand($reversePath);
452:     foreach ($recipients as $forwardPath)
453:     {
454:       try
455:       {
456:         $this->_doRcptToCommand($forwardPath);
457:         $sent++;
458:       }
459:       catch (Swift_TransportException $e)
460:       {
461:         $failedRecipients[] = $forwardPath;
462:       }
463:     }
464:     
465:     if ($sent != 0)
466:     {
467:       $this->_doDataCommand();
468:       $this->_streamMessage($message);
469:     }
470:     else
471:     {
472:       $this->reset();
473:     }
474:     
475:     return $sent;
476:   }

478:   /** Send a message to the given To: recipients */
479:   private function _sendTo(Swift_Mime_Message $message, $reversePath,
480:     array $to, array &$failedRecipients)
481:   {
482:     if (empty($to))
483:     {
484:       return 0;
485:     }
486:     return $this->_doMailTransaction($message, $reversePath, array_keys($to),
487:       $failedRecipients);
488:   }
489:   
490:   /** Send a message to the given Cc: recipients */
491:   private function _sendCc(Swift_Mime_Message $message, $reversePath,
492:     array $cc, array &$failedRecipients)
493:   {
494:     if (empty($cc))
495:     {
496:       return 0;
497:     }
498:     return $this->_doMailTransaction($message, $reversePath, array_keys($cc),
499:       $failedRecipients);
500:   }
501:   
502:   /** Send a message to all Bcc: recipients */
503:   private function _sendBcc(Swift_Mime_Message $message, $reversePath,
504:     array $bcc, array &$failedRecipients)
505:   {
506:     $sent = 0;
507:     foreach ($bcc as $forwardPath => $name)
508:     {
509:       $message->setBcc(array($forwardPath => $name));
510:       $sent += $this->_doMailTransaction(
511:         $message, $reversePath, array($forwardPath), $failedRecipients
512:         );
513:     }
514:     return $sent;
515:   }

517:   /** Try to determine the hostname of the server this is run on */
518:   private function _lookupHostname()
519:   {
520:     if (!empty($_SERVER['SERVER_NAME'])
521:       && $this->_isFqdn($_SERVER['SERVER_NAME']))
522:     {
523:       $this->_domain = $_SERVER['SERVER_NAME'];
524:     }
525:     elseif (!empty($_SERVER['SERVER_ADDR']))
526:     {
527:       $this->_domain = sprintf('[%s]', $_SERVER['SERVER_ADDR']);
528:     }
529:   }
530:   
531:   /** Determine is the $hostname is a fully-qualified name */
532:   private function _isFqdn($hostname)
533:   {
534:     //We could do a really thorough check, but there's really no point
535:     if (false !== $dotPos = strpos($hostname, '.'))
536:     {
537:       return ($dotPos > 0) && ($dotPos != strlen($hostname) - 1);
538:     }
539:     else
540:     {
541:       return false;
542:     }
543:   }
