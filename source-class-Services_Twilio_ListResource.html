 16:     public function __construct($client, $uri) {
 17:         $name = $this->getResourceName(true);
 18:         /* 
 19:          * By default trim the 's' from the end of the list name to get the
 20:          * instance name (ex Accounts -> Account). This behavior can be
 21:          * overridden by child classes if the rule doesn't work.
 22:          */
 23:         if (!isset($this->instance_name)) {
 24:             $this->instance_name = "Services_Twilio_Rest_" . rtrim($name, 's');
 25:         }
 26:         parent::__construct($client, $uri);
 27:     }
 28: 
 29:     /**
 30:      * Gets a resource from this list.
 31:      *
 32:      * @param string $sid The resource SID
 33:      * @return Services_Twilio_InstanceResource The resource
 34:      */
 35:     public function get($sid)
 36:     {
 37:         $instance = new $this->instance_name(
 38:             $this->client, $this->uri . "/$sid"
 39:         );
 40:         // XXX check if this is actually a sid in all cases.
 41:         $instance->sid = $sid;
 42:         return $instance;
 43:     }
 44: 
 45:     /* 
 46:      * Construct an InstanceResource with the specified params.
 47:      *
 48:      * @param array params usually a JSON HTTP response from the API
 49:      * @return Services_Twilio_InstanceResource An instance with properties 
 50:      *      initialized to the values in the params array.
 51:      */
 52:     public function getObjectFromJson($params, $idParam = "sid")
 53:     {
 54:         if (isset($params->{$idParam})) {
 55:             $uri = $this->uri . "/" . $params->{$idParam};
 56:         } else {
 57:             $uri = $this->uri;
 58:         }
 59:         return new $this->instance_name($this->client, $uri, $params);
 60:     }

 64:      *
 65:      * @param string $sid The resource SID
 66:      * @return null
 67:      */
 68:     public function delete($sid, array $params = array())
 69:     {
 70:         $this->client->deleteData($this->uri . '/' . $sid, $params);
 71:     }

 73:     /**
 74:      * Create a resource on the list and then return its representation as an

 77:      * @param array $params The parameters with which to create the resource
 78:      *
 79:      * @return Services_Twilio_InstanceResource The created resource
 80:      */
 81:     protected function _create(array $params)
 82:     {
 83:         $params = $this->client->createData($this->uri, $params);
 84:         /* Some methods like verified caller ID don't return sids. */
 85:         if (isset($params->sid)) {
 86:             $resource_uri = $this->uri . '/' . $params->sid;
 87:         } else {
 88:             $resource_uri = $this->uri;
 89:         }
 90:         return new $this->instance_name($this->client, $resource_uri, $params);
 91:     }

 93:     /**
 94:      * Returns a page of InstanceResources from this list.
 95:      *
 96:      * @param int    $page The start page
 97:      * @param int    $size Number of items per page
 98:      * @param array  $filters Optional filters
 99:      * @param string $deep_paging_uri if provided, the $page and $size 
100:      *      parameters will be ignored and this URI will be requested directly.
101:      *
102:      * @return Services_Twilio_Page A page
103:      */
104:     public function getPage(
105:         $page = 0, $size = 50, array $filters = array(), 
106:         $deep_paging_uri = null
107:     ) {
108:         $list_name = $this->getResourceName();
109:         if ($deep_paging_uri !== null) {
110:             $page = $this->client->retrieveData($deep_paging_uri, array(), true);
111:         } else {
112:             $page = $this->client->retrieveData($this->uri, array(
113:                 'Page' => $page,
114:                 'PageSize' => $size,
115:             ) + $filters);
116:         }
117: 
118:         /* create a new PHP object for each json obj in the api response. */
119:         $page->$list_name = array_map(
120:             array($this, 'getObjectFromJson'),
121:             $page->$list_name
122:         );
123:         if (isset($page->next_page_uri)) {
124:             $next_page_uri = $page->next_page_uri;
125:         } else {
126:             $next_page_uri = null;
127:         }
128:         return new Services_Twilio_Page($page, $list_name, $next_page_uri);
129:     }

156:     }
