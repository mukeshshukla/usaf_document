 45:  *
 46:  * @var object
 47:  * @access private
 48:  */

 50: /**
 51:  * settings
 52:  *      path = absolute path to cache directory, default => CACHE
 53:  *      prefix = string prefix for filename, default => cake_
 54:  *      lock = enable file locking on write, default => false
 55:  *      serialize = serialize the data, default => true
 56:  *
 57:  * @var array
 58:  * @see CacheEngine::__defaults
 59:  * @access public
 60:  */
 61:     var $settings = array();
 62: /**
 63:  * Set to true if FileEngine::init(); and FileEngine::__active(); do not fail.
 64:  *
 65:  * @var boolean
 66:  * @access private
 67:  */
 68:     var $__active = false;
 69: /**
 70:  * True unless FileEngine::__active(); fails
 71:  *
 72:  * @var boolean
 73:  * @access private
 74:  */
 75:     var $__init = true;
 76: /**
 77:  * Initialize the Cache Engine
 78:  *
 79:  * Called automatically by the cache frontend
 80:  * To reinitialize the settings call Cache::engine('EngineName', [optional] settings = array());
 81:  *
 82:  * @param array $setting array of setting for the engine
 83:  * @return boolean True if the engine has been successfully initialized, false if not
 84:  * @access public
 85:  */
 86:     function init($settings = array()) {
 87:         parent::init($settings);
 88:         $defaults = array('path' => CACHE, 'prefix'=> 'cake_', 'lock'=> false, 'serialize'=> true);
 89:         $this->settings = array_merge($defaults, $this->settings, $settings);
 90:         if(!isset($this->__File)) {
 91:             $this->__File =& new File($this->settings['path'] . DS . 'cake');
 92:         }
 93:         $this->settings['path'] = $this->__File->Folder->cd($this->settings['path']);
 94:         if(empty($this->settings['path'])) {
 95:             return false;
 96:         }
 97:         return $this->__active();
 98:     }
 99: /**
100:  * Garbage collection. Permanently remove all expired and deleted data
101:  *
102:  * @return boolean True if garbage collection was succesful, false on failure
103:  * @access public
104:  */
105:     function gc() {
106:         return $this->clear(true);
107:     }
108: /**
109:  * Write data for key into cache
110:  *
111:  * @param string $key Identifier for the data
112:  * @param mixed $data Data to be cached
113:  * @param mixed $duration How long to cache the data, in seconds
114:  * @return boolean True if the data was succesfully cached, false on failure
115:  * @access public
116:  */
117:     function write($key, &$data, $duration) {
118:         if (empty($data) || !$this->__init) {
119:             return false;
120:         }
121: 
122:         if($this->__setKey($key) === false) {
123:             return false;
124:         }
125: 
126:         if ($duration == null) {
127:             $duration = $this->settings['duration'];
128:         }
129:         $windows = false;
130:         $lineBreak = "\n";
131: 
132:         if (substr(PHP_OS, 0, 3) == "WIN") {
133:             $lineBreak = "\r\n";
134:             $windows = true;
135:         }
136:         if (!empty($this->settings['serialize'])) {
137:             if ($windows) {
138:                 $data = str_replace('\\', '\\\\\\\\', serialize($data));
139:             } else {
140:                 $data = serialize($data);
141:             }
142:         }
143: 
144:         if ($this->settings['lock']) {
145:             $this->__File->lock = true;
146:         }
147:         $expires = time() + $duration;
148:         $contents = $expires . $lineBreak . $data . $lineBreak;
149:         $success = $this->__File->write($contents);
150:         $this->__File->close();
151:         return $success;
152:     }
153: /**
154:  * Read a key from the cache
155:  *
156:  * @param string $key Identifier for the data
157:  * @return mixed The cached data, or false if the data doesn't exist, has expired, or if there was an error fetching it
158:  * @access public
159:  */
160:     function read($key) {
161:         if($this->__setKey($key) === false || !$this->__init) {
162:             return false;
163:         }
164:         if ($this->settings['lock']) {
165:             $this->__File->lock = true;
166:         }
167:         $cachetime = $this->__File->read(11);
168: 
169:         if ($cachetime !== false && intval($cachetime) < time()) {
170:             $this->__File->close();
171:             $this->__File->delete();
172:             return false;
173:         }
174:         $data = $this->__File->read(true);
175: 
176:         if (!empty($data) && !empty($this->settings['serialize'])) {
177:             $data = stripslashes($data);
178:             $data = preg_replace('!s:(\d+):"(.*?)";!se', "'s:'.strlen('$2').':\"$2\";'", $data);
179:             $data = unserialize($data);
180: 
181:             if (is_array($data)) {
182:                 $data = array_map('stripslashes_deep', $data);
183:             }
184:         }
185:         $this->__File->close();
186:         return $data;
187:     }
188: /**
189:  * Delete a key from the cache
190:  *
191:  * @param string $key Identifier for the data
192:  * @return boolean True if the value was successfully deleted, false if it didn't exist or couldn't be removed
193:  * @access public
194:  */
195:     function delete($key) {
196:         if($this->__setKey($key) === false || !$this->__init) {
197:             return false;
198:         }
199:         return $this->__File->delete();
200:     }
201: /**
202:  * Delete all values from the cache
203:  *
204:  * @param boolean $check Optional - only delete expired cache items
205:  * @return boolean True if the cache was succesfully cleared, false otherwise
206:  * @access public
207:  */
208:     function clear($check) {
209:         if (!$this->__init) {
210:             return false;
211:         }
212:         $dir = dir($this->settings['path']);
213:         if ($check) {
214:             $now = time();
215:             $threshold = $now - $this->settings['duration'];
216:         }
217:         while (($entry = $dir->read()) !== false) {
218:             if($this->__setKey(str_replace($this->settings['prefix'], '', $entry)) === false) {
219:                 continue;
220:             }
221:             if ($check) {
222:                 $mtime = $this->__File->lastChange();
223: 
224:                 if ($mtime === false || $mtime > $threshold) {
225:                     continue;
226:                 }
227: 
228:                 $expires = $this->__File->read(11);
229:                 $this->__File->close();
230: 
231:                 if ($expires > $now) {
232:                     continue;
233:                 }
234:             }
235:             $this->__File->delete();
236:         }
237:         $dir->close();
238:         return true;
239:     }
240: /**
241:  * Get absolute file for a given key
242:  *
243:  * @param string $key The key
244:  * @return mixed Absolute cache file for the given key or false if erroneous
245:  * @access private
246:  */
247:     function __setKey($key) {
248:         $this->__File->Folder->cd($this->settings['path']);
249:         $this->__File->name = $this->settings['prefix'] . $key;
250:         if (!$this->__File->Folder->inPath($this->__File->pwd(), true)) {
251:             return false;
252:         }
253:     }
254: /**
255:  * Determine is cache directory is writable
256:  *
257:  * @return boolean
258:  * @access private
259:  */
260:     function __active() {
261:         if (!$this->__active && $this->__init && !is_writable($this->settings['path'])) {
262:             $this->__init = false;
263:             trigger_error(sprintf(__('%s is not writable', true), $this->settings['path']), E_USER_WARNING);
264:         } else {
265:             $this->__active = true;
266:         }
267:         return true;
268:     }
