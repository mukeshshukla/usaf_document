 24: /**
 25:  * Provides a mechanism for storing data using two keys.
 26:  * @package Swift
 27:  * @subpackage KeyCache
 28:  * @author Chris Corbyn
 29:  */
 30: interface Swift_KeyCache
 31: {
 32:   
 33:   /** Mode for replacing existing cached data */
 34:   const MODE_WRITE = 1;
 35:   
 36:   /** Mode for appending data to the end of existing cached data */
 37:   const MODE_APPEND = 2;
 38:   
 39:   /**
 40:    * Set a string into the cache under $itemKey for the namespace $nsKey.
 41:    * @param string $nsKey
 42:    * @param string $itemKey
 43:    * @param string $string
 44:    * @param int $mode
 45:    * @see MODE_WRITE, MODE_APPEND
 46:    */
 47:   public function setString($nsKey, $itemKey, $string, $mode);
 48:   
 49:   /**
 50:    * Set a ByteStream into the cache under $itemKey for the namespace $nsKey.
 51:    * @param string $nsKey
 52:    * @param string $itemKey
 53:    * @param Swift_OutputByteStream $os
 54:    * @param int $mode
 55:    * @see MODE_WRITE, MODE_APPEND
 56:    */
 57:   public function importFromByteStream($nsKey, $itemKey, Swift_OutputByteStream $os,
 58:     $mode);
 59:   
 60:   /**
 61:    * Provides a ByteStream which when written to, writes data to $itemKey.
 62:    * NOTE: The stream will always write in append mode.
 63:    * If the optional third parameter is passed all writes will go through $is.
 64:    * @param string $nsKey
 65:    * @param string $itemKey
 66:    * @param Swift_InputByteStream $is, optional
 67:    * @return Swift_InputByteStream
 68:    */
 69:   public function getInputByteStream($nsKey, $itemKey,
 70:     Swift_InputByteStream $is = null);
 71:   
 72:   /**
 73:    * Get data back out of the cache as a string.
 74:    * @param string $nsKey
 75:    * @param string $itemKey
 76:    * @return string
 77:    */
 78:   public function getString($nsKey, $itemKey);
 79:   
 80:   /**
 81:    * Get data back out of the cache as a ByteStream.
 82:    * @param string $nsKey
 83:    * @param string $itemKey
 84:    * @param Swift_InputByteStream $is to write the data to
 85:    */
 86:   public function exportToByteStream($nsKey, $itemKey, Swift_InputByteStream $is);
 87:   
 88:   /**
 89:    * Check if the given $itemKey exists in the namespace $nsKey.
 90:    * @param string $nsKey
 91:    * @param string $itemKey
 92:    * @return boolean
 93:    */
 94:   public function hasKey($nsKey, $itemKey);
 95:   
 96:   /**
 97:    * Clear data for $itemKey in the namespace $nsKey if it exists.
 98:    * @param string $nsKey
 99:    * @param string $itemKey
100:    */
101:   public function clearKey($nsKey, $itemKey);
102:   
103:   /**
104:    * Clear all data in the namespace $nsKey if it exists.
105:    * @param string $nsKey
106:    */
107:   public function clearAll($nsKey);
108:   
109: }
