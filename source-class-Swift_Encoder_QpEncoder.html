115:   /**
116:    * Creates a new QpEncoder for the given CharacterStream.
117:    * @param Swift_CharacterStream $charStream to use for reading characters
118:    * @param Swift_StreamFilter $filter if input should be canonicalized
119:    */
120:   public function __construct(Swift_CharacterStream $charStream,
121:     Swift_StreamFilter $filter = null)
122:   {
123:     $this->_charStream = $charStream;
124:     if (empty(self::$_safeMap))
125:     {
126:       foreach (array_merge(
127:         array(0x09, 0x20), range(0x21, 0x3C), range(0x3E, 0x7E)) as $byte)
128:       {
129:         self::$_safeMap[$byte] = chr($byte);
130:       }
131:     }
132:     $this->_filter = $filter;
133:   }
134: 
135:   /**
136:    * Takes an unencoded string and produces a QP encoded string from it.
137:    * QP encoded strings have a maximum line length of 76 characters.
138:    * If the first line needs to be shorter, indicate the difference with
139:    * $firstLineOffset.
140:    * @param string $string to encode
141:    * @param int $firstLineOffset, optional
142:    * @param int $maxLineLength, optional, 0 indicates the default of 76 chars
143:    * @return string
144:    */
145:   public function encodeString($string, $firstLineOffset = 0,
146:     $maxLineLength = 0)
147:   {
148:     if ($maxLineLength > 76 || $maxLineLength <= 0)
149:     {
150:       $maxLineLength = 76;
151:     }
152: 
153:     $thisLineLength = $maxLineLength - $firstLineOffset;
154: 
155:     $lines = array();
156:     $lNo = 0;
157:     $lines[$lNo] = '';
158:     $currentLine =& $lines[$lNo++];
159:     $size=$lineLen=0;
160: 
161:     $this->_charStream->flushContents();
162:     $this->_charStream->importString($string);
163: 
164:     //Fetching more than 4 chars at one is slower, as is fetching fewer bytes
165:     // Conveniently 4 chars is the UTF-8 safe number since UTF-8 has up to 6
166:     // bytes per char and (6 * 4 * 3 = 72 chars per line) * =NN is 3 bytes
167:     while (false !== $bytes = $this->_nextSequence())
168:     {
169:       //If we're filtering the input
170:       if (isset($this->_filter))
171:       {
172:         //If we can't filter because we need more bytes
173:         while ($this->_filter->shouldBuffer($bytes))
174:         {
175:           //Then collect bytes into the buffer
176:           if (false === $moreBytes = $this->_nextSequence(1))
177:           {
178:             break;
179:           }
180: 
181:           foreach ($moreBytes as $b)
182:           {
183:             $bytes[] = $b;
184:           }
185:         }
186:         //And filter them
187:         $bytes = $this->_filter->filter($bytes);
188:       }
189: 
190:       $enc = $this->_encodeByteSequence($bytes, $size);
191:       if ($currentLine && $lineLen+$size >= $thisLineLength)
192:       {
193:         $lines[$lNo] = '';
194:         $currentLine =& $lines[$lNo++];
195:         $thisLineLength = $maxLineLength;
196:         $lineLen=0;
197:       }
198:       $lineLen+=$size;
199:       $currentLine .= $enc;
200:     }
201: 
202:     return $this->_standardize(implode("=\r\n", $lines));
203:   }

223:   {
224:     $ret = '';
225:     $size=0;
226:     foreach ($bytes as $b)
227:     {
228:       if (isset(self::$_safeMap[$b]))
229:       {
230:         $ret .= self::$_safeMap[$b];
231:         ++$size;
232:       }
233:       else
234:       {
235:         $ret .= self::$_qpMap[$b];
236:         $size+=3;
237:       }
238:     }
239:     return $ret;

267:       case 0x20:

269:     }
270:     return $string;
