23: /**
24:  * Processes bytes as they pass through a buffer and replaces sequences in it.
25:  * @package Swift
26:  * @author Chris Corbyn
27:  */
28: class Swift_StreamFilters_StringReplacementFilter implements Swift_StreamFilter
29: {
30:   
31:   /** The needle(s) to search for */
32:   private $_search;
33:   
34:   /** The replacement(s) to make */
35:   private $_replace;
36:   
37:   /**
38:    * Create a new StringReplacementFilter with $search and $replace.
39:    * @param string|array $search
40:    * @param string|array $replace
41:    */
42:   public function __construct($search, $replace)
43:   {
44:     $this->_search = $search;
45:     $this->_replace = $replace;
46:   }
47:   
48:   /**
49:    * Returns true if based on the buffer passed more bytes should be buffered.
50:    * @param string $buffer
51:    * @return boolean
52:    */
53:   public function shouldBuffer($buffer)
54:   {
55:     $endOfBuffer = substr($buffer, -1);
56:     foreach ((array) $this->_search as $needle)
57:     {
58:       if (false !== strpos($needle, $endOfBuffer))
59:       {
60:         return true;
61:       }
62:     }
63:     return false;
64:   }
65:   
66:   /**
67:    * Perform the actual replacements on $buffer and return the result.
68:    * @param string $buffer
69:    * @return string
70:    */
71:   public function filter($buffer)
72:   {
73:     return str_replace($this->_search, $this->_replace, $buffer);
74:   }
75:   
76: }
