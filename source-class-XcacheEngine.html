 36: /**
 37:  * settings
 38:  *      PHP_AUTH_USER = xcache.admin.user, default cake
 39:  *      PHP_AUTH_PW = xcache.admin.password, default cake
 40:  *
 41:  * @var array
 42:  * @access public
 43:  */
 44:     var $settings = array();
 45: /**
 46:  * Initialize the Cache Engine
 47:  *
 48:  * Called automatically by the cache frontend
 49:  * To reinitialize the settings call Cache::engine('EngineName', [optional] settings = array());
 50:  *
 51:  * @param array $setting array of setting for the engine
 52:  * @return boolean True if the engine has been successfully initialized, false if not
 53:  * @access public
 54:  */
 55:     function init($settings) {
 56:         parent::init($settings);
 57:         $defaults = array('PHP_AUTH_USER' => 'cake', 'PHP_AUTH_PW' => 'cake');
 58:         $this->settings = array_merge($this->settings, $defaults, $settings);
 59:         return function_exists('xcache_info');
 60:     }
 61: /**
 62:  * Write data for key into cache
 63:  *
 64:  * @param string $key Identifier for the data
 65:  * @param mixed $value Data to be cached
 66:  * @param integer $duration How long to cache the data, in seconds
 67:  * @return boolean True if the data was succesfully cached, false on failure
 68:  * @access public
 69:  */
 70:     function write($key, &$value, $duration) {
 71:         return xcache_set($key, $value, $duration);
 72:     }
 73: /**
 74:  * Read a key from the cache
 75:  *
 76:  * @param string $key Identifier for the data
 77:  * @return mixed The cached data, or false if the data doesn't exist, has expired, or if there was an error fetching it
 78:  * @access public
 79:  */
 80:     function read($key) {
 81:         if (xcache_isset($key)) {
 82:             return xcache_get($key);
 83:         }
 84:         return false;
 85:     }
 86: /**
 87:  * Delete a key from the cache
 88:  *
 89:  * @param string $key Identifier for the data
 90:  * @return boolean True if the value was succesfully deleted, false if it didn't exist or couldn't be removed
 91:  * @access public
 92:  */
 93:     function delete($key) {
 94:         return xcache_unset($key);
 95:     }
 96: /**
 97:  * Delete all keys from the cache
 98:  *
 99:  * @return boolean True if the cache was succesfully cleared, false otherwise
100:  * @access public
101:  */
102:     function clear() {
103:         $result = true;
104:         $this->__auth();
105:         for ($i = 0, $max = xcache_count(XC_TYPE_VAR); $i < $max; $i++) {
106:             if (!xcache_clear_cache(XC_TYPE_VAR, $i)) {
107:                 $result = false;
108:                 break;
109:             }
110:         }
111:         $this->__auth(true);
112:         return $result;
113:     }
114: /**
115:  * Populates and reverses $_SERVER authentication values
116:  * Makes necessary changes (and reverting them back) in $_SERVER
117:  *
118:  * This has to be done because xcache_clear_cache() needs to pass Basic Http Auth
119:  * (see xcache.admin configuration settings)
120:  *
121:  * @param boolean Revert changes
122:  * @access private
123:  */
124:     function __auth($reverse = false) {
125:         static $backup = array();
126:         $keys = array('PHP_AUTH_USER', 'PHP_AUTH_PW');
127:         foreach ($keys as $key) {
128:             if ($reverse) {
129:                 if (isset($backup[$key])) {
130:                     $_SERVER[$key] = $backup[$key];
131:                     unset($backup[$key]);
132:                 } else {
133:                     unset($_SERVER[$key]);
134:                 }
135:             } else {
136:                 $value = env($key);
137:                 if (!empty($value)) {
138:                     $backup[$key] = $value;
139:                 }
140:                 $varName = '__' . $key;
141:                 $_SERVER[$key] = $this->settings[$varName];
142:             }
143:         }
144:     }
145: }
