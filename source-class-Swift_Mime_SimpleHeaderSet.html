 25:  * A collection of MIME headers.
 26:  * 
 27:  * @package Swift
 28:  * @subpackage Mime

 30:  * @author Chris Corbyn
 31:  */

 33: {
 34:   
 35:   /** HeaderFactory */
 36:   private $_factory;
 37:   
 38:   /** Collection of set Headers */
 39:   private $_headers = array();
 40:   
 41:   /** Field ordering details */
 42:   private $_order = array();
 43:   
 44:   /** List of fields which are required to be displayed */
 45:   private $_required = array();
 46:   
 47:   /** The charset used by Headers */
 48:   private $_charset;
 49:   
 50:   /**
 51:    * Create a new SimpleHeaderSet with the given $factory.
 52:    * 
 53:    * @param Swift_Mime_HeaderFactory $factory
 54:    * @param string $charset
 55:    */
 56:   public function __construct(Swift_Mime_HeaderFactory $factory,
 57:     $charset = null)
 58:   {
 59:     $this->_factory = $factory;
 60:     if (isset($charset))
 61:     {
 62:       $this->setCharset($charset);
 63:     }
 64:   }
 65:   
 66:   /**
 67:    * Set the charset used by these headers.
 68:    * 
 69:    * @param string $charset
 70:    */
 71:   public function setCharset($charset)
 72:   {
 73:     $this->_charset = $charset;
 74:     $this->_factory->charsetChanged($charset);
 75:     $this->_notifyHeadersOfCharset($charset);
 76:   }
 77:   
 78:   /**
 79:    * Add a new Mailbox Header with a list of $addresses.
 80:    * 
 81:    * @param string $name
 82:    * @param array|string $addresses
 83:    */
 84:   public function addMailboxHeader($name, $addresses = null)
 85:   {
 86:     $this->_storeHeader($name,
 87:       $this->_factory->createMailboxHeader($name, $addresses));
 88:   }
 89:   
 90:   /**
 91:    * Add a new Date header using $timestamp (UNIX time).
 92:    * 
 93:    * @param string $name
 94:    * @param int $timestamp
 95:    */
 96:   public function addDateHeader($name, $timestamp = null)
 97:   {
 98:     $this->_storeHeader($name,
 99:       $this->_factory->createDateHeader($name, $timestamp));
100:   }
101:   
102:   /**
103:    * Add a new basic text header with $name and $value.
104:    * 
105:    * @param string $name
106:    * @param string $value
107:    */
108:   public function addTextHeader($name, $value = null)
109:   {
110:     $this->_storeHeader($name,
111:       $this->_factory->createTextHeader($name, $value));
112:   }
113:   
114:   /**
115:    * Add a new ParameterizedHeader with $name, $value and $params.
116:    * 
117:    * @param string $name
118:    * @param string $value
119:    * @param array $params
120:    */
121:   public function addParameterizedHeader($name, $value = null,
122:     $params = array())
123:   {
124:     $this->_storeHeader($name,
125:       $this->_factory->createParameterizedHeader($name, $value,
126:       $params));
127:   }
128:   
129:   /**
130:    * Add a new ID header for Message-ID or Content-ID.
131:    * 
132:    * @param string $name
133:    * @param string|array $ids
134:    */
135:   public function addIdHeader($name, $ids = null)
136:   {
137:     $this->_storeHeader($name, $this->_factory->createIdHeader($name, $ids));
138:   }
139:   
140:   /**
141:    * Add a new Path header with an address (path) in it.
142:    * 
143:    * @param string $name
144:    * @param string $path
145:    */
146:   public function addPathHeader($name, $path = null)
147:   {
148:     $this->_storeHeader($name, $this->_factory->createPathHeader($name, $path));
149:   }
150:   
151:   /**
152:    * Returns true if at least one header with the given $name exists.
153:    * 
154:    * If multiple headers match, the actual one may be specified by $index.
155:    * 
156:    * @param string $name
157:    * @param int $index
158:    * 
159:    * @return boolean
160:    */
161:   public function has($name, $index = 0)
162:   {
163:     $lowerName = strtolower($name);
164:     return array_key_exists($lowerName, $this->_headers)
165:       && array_key_exists($index, $this->_headers[$lowerName]);
166:   }
167:   
168:   /**
169:    * Set a header in the HeaderSet.
170:    * 
171:    * The header may be a previously fetched header via {@link get()} or it may
172:    * be one that has been created separately.
173:    * 
174:    * If $index is specified, the header will be inserted into the set at this
175:    * offset.
176:    * 
177:    * @param Swift_Mime_Header $header
178:    * @param int $index
179:    */
180:   public function set(Swift_Mime_Header $header, $index = 0)
181:   {
182:     $this->_storeHeader($header->getFieldName(), $header, $index);
183:   }
184:   
185:   /**
186:    * Get the header with the given $name.
187:    * 
188:    * If multiple headers match, the actual one may be specified by $index.
189:    * Returns NULL if none present.
190:    * 
191:    * @param string $name
192:    * @param int $index
193:    * 
194:    * @return Swift_Mime_Header
195:    */
196:   public function get($name, $index = 0)
197:   {
198:     if ($this->has($name, $index))
199:     {
200:       $lowerName = strtolower($name);
201:       return $this->_headers[$lowerName][$index];
202:     }
203:   }
204:   
205:   /**
206:    * Get all headers with the given $name.
207:    * 
208:    * @param string $name
209:    * 
210:    * @return array
211:    */
212:   public function getAll($name = null)
213:   {
214:     if (!isset($name))
215:     {
216:       $headers = array();
217:       foreach ($this->_headers as $collection)
218:       {
219:         $headers = array_merge($headers, $collection);
220:       }
221:       return $headers;
222:     }
223:     
224:     $lowerName = strtolower($name);
225:     if (!array_key_exists($lowerName, $this->_headers))
226:     {
227:       return array();
228:     }
229:     return $this->_headers[$lowerName];
230:   }
231:   
232:   /**
233:    * Remove the header with the given $name if it's set.
234:    * 
235:    * If multiple headers match, the actual one may be specified by $index.
236:    * 
237:    * @param string $name
238:    * @param int $index
239:    */
240:   public function remove($name, $index = 0)
241:   {
242:     $lowerName = strtolower($name);
243:     unset($this->_headers[$lowerName][$index]);
244:   }
245:   
246:   /**
247:    * Remove all headers with the given $name.
248:    * 
249:    * @param string $name
250:    */
251:   public function removeAll($name)
252:   {
253:     $lowerName = strtolower($name);
254:     unset($this->_headers[$lowerName]);
255:   }
256:   
257:   /**
258:    * Create a new instance of this HeaderSet.
259:    * 
260:    * @return Swift_Mime_HeaderSet
261:    */
262:   public function newInstance()
263:   {
264:     return new self($this->_factory);
265:   }
266:   
267:   /**
268:    * Define a list of Header names as an array in the correct order.
269:    * 
270:    * These Headers will be output in the given order where present.
271:    * 
272:    * @param array $sequence
273:    */
274:   public function defineOrdering(array $sequence)
275:   {
276:     $this->_order = array_flip(array_map('strtolower', $sequence));
277:   }
278:   
279:   /**
280:    * Set a list of header names which must always be displayed when set.
281:    * 
282:    * Usually headers without a field value won't be output unless set here.
283:    * 
284:    * @param array $names
285:    */
286:   public function setAlwaysDisplayed(array $names)
287:   {
288:     $this->_required = array_flip(array_map('strtolower', $names));
289:   }
290: 
291:   /**
292:    * Notify this observer that the entity's charset has changed.
293:    * 
294:    * @param string $charset
295:    */
296:   public function charsetChanged($charset)
297:   {
298:     $this->setCharset($charset);
299:   }
300:   
301:   /**
302:    * Returns a string with a representation of all headers.
303:    * 
304:    * @return string
305:    */
306:   public function toString()
307:   {
308:     $string = '';
309:     $headers = $this->_headers;
310:     if ($this->_canSort())
311:     {
312:       uksort($headers, array($this, '_sortHeaders'));
313:     }
314:     foreach ($headers as $collection)
315:     {
316:       foreach ($collection as $header)
317:       {
318:         if ($this->_isDisplayed($header) || $header->getFieldBody() != '')
319:         {
320:           $string .= $header->toString();
321:         }
322:       }
323:     }
324:     return $string;
325:   }
326:   
327:   // -- Private methods
328:   
329:   /** Save a Header to the internal collection */
330:   private function _storeHeader($name, Swift_Mime_Header $header, $offset = null)
331:   {
332:     if (!isset($this->_headers[strtolower($name)]))
333:     {
334:       $this->_headers[strtolower($name)] = array();
335:     }
336:     if (!isset($offset))
337:     {
338:       $this->_headers[strtolower($name)][] = $header;
339:     }
340:     else
341:     {
342:       $this->_headers[strtolower($name)][$offset] = $header;
343:     }
344:   }
345:   
346:   /** Test if the headers can be sorted */
347:   private function _canSort()
348:   {
349:     return count($this->_order) > 0;
350:   }
351:   
352:   /** uksort() algorithm for Header ordering */
353:   private function _sortHeaders($a, $b)
354:   {
355:     $lowerA = strtolower($a);
356:     $lowerB = strtolower($b);
357:     $aPos = array_key_exists($lowerA, $this->_order)
358:       ? $this->_order[$lowerA]
359:       : -1;
360:     $bPos = array_key_exists($lowerB, $this->_order)
361:       ? $this->_order[$lowerB]
362:       : -1;
363:       
364:     if ($aPos == -1)
365:     {
366:       return 1;
367:     }
368:     elseif ($bPos == -1)
369:     {
370:       return -1;
371:     }
372:     
373:     return ($aPos < $bPos) ? -1 : 1;
374:   }
375:   
376:   /** Test if the given Header is always displayed */
377:   private function _isDisplayed(Swift_Mime_Header $header)
378:   {
379:     return array_key_exists(strtolower($header->getFieldName()), $this->_required);
380:   }
381:   
382:   /** Notify all Headers of the new charset */
383:   private function _notifyHeadersOfCharset($charset)
384:   {
385:     foreach ($this->_headers as $headerGroup)
386:     {
387:       foreach ($headerGroup as $header)
388:       {
389:         $header->setCharset($charset);
390:       }
391:     }
392:   }
