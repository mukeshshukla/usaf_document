327: /**
328:  * Generates the fields list of an SQL query.

334:  */

336:         if (empty($alias)) {
337:             $alias = $model->alias;
338:         }
339: 
340:         if (!is_array($fields)) {
341:             if ($fields != null) {
342:                 if (strpos($fields, ',')) {
343:                     $fields = explode(',', $fields);
344:                 } else {
345:                     $fields = array($fields);
346:                 }
347:                 $fields = array_map('trim', $fields);
348:             } else {
349:                 $fields = array_keys($model->schema());
350:             }
351:         }
352: 
353:         $count = count($fields);
354: 
355:         if ($count >= 1 && $fields[0] != '*' && strpos($fields[0], 'COUNT(*)') === false) {
356:             for ($i = 0; $i < $count; $i++) {
357:                 if (!preg_match('/^.+\\(.*\\)/', $fields[$i])) {
358:                     $prepend = '';
359:                     if (strpos($fields[$i], 'DISTINCT') !== false) {
360:                         $prepend = 'DISTINCT ';

362:                     }
363: 
364:                     $dot = strrpos($fields[$i], '.');
365:                     if ($dot === false) {
366:                         $fields[$i] = $prepend . $this->name($alias) . '.' . $this->name($fields[$i]) . ' AS ' . $this->name($alias . '__' . $fields[$i]);
