 25: /**
 26:  * A MIME entity, such as an attachment.
 27:  * @package Swift
 28:  * @subpackage Mime
 29:  * @author Chris Corbyn
 30:  */
 31: interface Swift_Mime_MimeEntity
 32:   extends Swift_Mime_CharsetObserver, Swift_Mime_EncodingObserver
 33: {
 34:   
 35:   /** Main message document; there can only be one of these */
 36:   const LEVEL_TOP = 16;
 37:   
 38:   /** An entity which nests with the same precedence as an attachment */
 39:   const LEVEL_MIXED = 256;
 40:   
 41:   /** An entity which nests with the same precedence as a mime part */
 42:   const LEVEL_ALTERNATIVE = 4096;
 43:   
 44:   /** An entity which nests with the same precedence as embedded content */
 45:   const LEVEL_RELATED = 65536;
 46:   
 47:   /**
 48:    * Get the level at which this entity shall be nested in final document.
 49:    * The lower the value, the more outermost the entity will be nested.
 50:    * @return int
 51:    * @see LEVEL_TOP, LEVEL_MIXED, LEVEL_RELATED, LEVEL_ALTERNATIVE
 52:    */
 53:   public function getNestingLevel();
 54:   
 55:   /**
 56:    * Get the qualified content-type of this mime entity.
 57:    * @return string
 58:    */
 59:   public function getContentType();
 60:   
 61:   /**
 62:    * Returns a unique ID for this entity.
 63:    * For most entities this will likely be the Content-ID, though it has
 64:    * no explicit semantic meaning and can be considered an identifier for
 65:    * programming logic purposes.
 66:    * If a Content-ID header is present, this value SHOULD match the value of
 67:    * the header.
 68:    * @return string
 69:    */
 70:   public function getId();
 71:   
 72:   /**
 73:    * Get all children nested inside this entity.
 74:    * These are not just the immediate children, but all children.
 75:    * @return Swift_Mime_MimeEntity[]
 76:    */
 77:   public function getChildren();
 78:   
 79:   /**
 80:    * Set all children nested inside this entity.
 81:    * This includes grandchildren.
 82:    * @param Swift_Mime_MimeEntity[] $children
 83:    */
 84:   public function setChildren(array $children);
 85:   
 86:   /**
 87:    * Get the collection of Headers in this Mime entity.
 88:    * @return Swift_Mime_Header[]
 89:    */
 90:   public function getHeaders();
 91:   
 92:   /**
 93:    * Get the body content of this entity as a string.
 94:    * Returns NULL if no body has been set.
 95:    * @return string
 96:    */
 97:   public function getBody();
 98:   
 99:   /**
100:    * Set the body content of this entity as a string.
101:    * @param string $body
102:    * @param string $contentType optional
103:    */
104:   public function setBody($body, $contentType = null);
105:   
106:   /**
107:    * Get this entire entity in its string form.
108:    * @return string
109:    */
110:   public function toString();
111:   
112:   /**
113:    * Get this entire entity as a ByteStream.
114:    * @param Swift_InputByteStream $is to write to
115:    */
116:   public function toByteStream(Swift_InputByteStream $is);
117:   
118: }
