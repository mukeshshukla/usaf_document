 24: /**
 25:  * An ID MIME Header for something like Message-ID or Content-ID.
 26:  * @package Swift
 27:  * @subpackage Mime
 28:  * @author Chris Corbyn
 29:  */
 30: class Swift_Mime_Headers_IdentificationHeader
 31:   extends Swift_Mime_Headers_AbstractHeader
 32: {
 33:   
 34:   /**
 35:    * The IDs used in the value of this Header.
 36:    * This may hold multiple IDs or just a single ID.
 37:    * @var string[]
 38:    * @access private
 39:    */
 40:   private $_ids = array();
 41:   
 42:   /**
 43:    * Creates a new IdentificationHeader with the given $name and $id.
 44:    * @param string $name
 45:    */
 46:   public function __construct($name)
 47:   {
 48:     $this->setFieldName($name);
 49:     $this->initializeGrammar();
 50:   }
 51:   
 52:   /**
 53:    * Get the type of Header that this instance represents.
 54:    * @return int
 55:    * @see TYPE_TEXT, TYPE_PARAMETERIZED, TYPE_MAILBOX
 56:    * @see TYPE_DATE, TYPE_ID, TYPE_PATH
 57:    */
 58:   public function getFieldType()
 59:   {
 60:     return self::TYPE_ID;
 61:   }
 62:   
 63:   /**
 64:    * Set the model for the field body.
 65:    * This method takes a string ID, or an array of IDs
 66:    * @param mixed $model
 67:    * @throws Swift_RfcComplianceException
 68:    */
 69:   public function setFieldBodyModel($model)
 70:   {
 71:     $this->setId($model);
 72:   }
 73:   
 74:   /**
 75:    * Get the model for the field body.
 76:    * This method returns an array of IDs
 77:    * @return array
 78:    */
 79:   public function getFieldBodyModel()
 80:   {
 81:     return $this->getIds();
 82:   }
 83:   
 84:   /**
 85:    * Set the ID used in the value of this header.
 86:    * @param string $id
 87:    * @throws Swift_RfcComplianceException
 88:    */
 89:   public function setId($id)
 90:   {
 91:     return $this->setIds(array($id));
 92:   }
 93:   
 94:   /**
 95:    * Get the ID used in the value of this Header.
 96:    * If multiple IDs are set only the first is returned.
 97:    * @return string
 98:    */
 99:   public function getId()
100:   {
101:     if (count($this->_ids) > 0)
102:     {
103:       return $this->_ids[0];
104:     }
105:   }
106:   
107:   /**
108:    * Set a collection of IDs to use in the value of this Header.
109:    * @param string[] $ids
110:    * @throws Swift_RfcComplianceException
111:    */
112:   public function setIds(array $ids)
113:   {
114:     $actualIds = array();
115:     
116:     foreach ($ids as $k => $id)
117:     {
118:       if (preg_match(
119:         '/^' . $this->getGrammar('id-left') . '@' .
120:         $this->getGrammar('id-right') . '$/D',
121:         $id
122:         ))
123:       {
124:         $actualIds[] = $id;
125:       }
126:       else
127:       {
128:         throw new Swift_RfcComplianceException(
129:           'Invalid ID given <' . $id . '>'
130:           );
131:       }
132:     }
133:     
134:     $this->clearCachedValueIf($this->_ids != $actualIds);
135:     $this->_ids = $actualIds;
136:   }
137:   
138:   /**
139:    * Get the list of IDs used in this Header.
140:    * @return string[]
141:    */
142:   public function getIds()
143:   {
144:     return $this->_ids;
145:   }
146:   
147:   /**
148:    * Get the string value of the body in this Header.
149:    * This is not necessarily RFC 2822 compliant since folding white space will
150:    * not be added at this stage (see {@link toString()} for that).
151:    * @return string
152:    * @see toString()
153:    * @throws Swift_RfcComplianceException
154:    */
155:   public function getFieldBody()
156:   {
157:     if (!$this->getCachedValue())
158:     {
159:       $angleAddrs = array();
160:     
161:       foreach ($this->_ids as $id)
162:       {
163:         $angleAddrs[] = '<' . $id . '>';
164:       }
165:     
166:       $this->setCachedValue(implode(' ', $angleAddrs));
167:     }
168:     return $this->getCachedValue();
169:   }
170:   
171: }
