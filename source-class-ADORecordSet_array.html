 725:     /**

 727:         Advantages include:
 728:         
 729:         a. StartTrans/CompleteTrans is nestable, unlike BeginTrans/CommitTrans/RollbackTrans.
 730:            Only the outermost block is treated as a transaction.<br>
 731:         b. CompleteTrans auto-detects SQL errors, and will rollback on errors, commit otherwise.<br>
 732:         c. All BeginTrans/CommitTrans/RollbackTrans inside a StartTrans/CompleteTrans block
 733:            are disabled, making it backward compatible.
 734:     */
 735:     function StartTrans($errfn = 'ADODB_TransMonitor')
 736:     {
 737:         if ($this->transOff > 0) {
 738:             $this->transOff += 1;
 739:             return;
 740:         }
 741:         
 742:         $this->_oldRaiseFn = $this->raiseErrorFn;
 743:         $this->raiseErrorFn = $errfn;
 744:         $this->_transOK = true;
 745:         
 746:         if ($this->debug && $this->transCnt > 0) ADOConnection::outp("Bad Transaction: StartTrans called within BeginTrans");
 747:         $this->BeginTrans();

 754:         for sql errors, and will commit or rollback as appropriate.
 755:         
 756:         @autoComplete if true, monitor sql errors and commit and rollback as appropriate, 
 757:         and if set to false force rollback even if no SQL error detected.
 758:         @returns true on commit, false on rollback.
 759:     */
 760:     function CompleteTrans($autoComplete = true)
 761:     {

 765:         }

 768:         $this->transOff = 0;

 773:             } else

 775:         } else {
