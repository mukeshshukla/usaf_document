 23: /**
 24:  * Dependency Injection container.
 25:  * @package Swift
 26:  * @author Chris Corbyn
 27:  */
 28: class Swift_DependencyContainer
 29: {
 30:   
 31:   /** Constant for literal value types */
 32:   const TYPE_VALUE = 0x0001;
 33:   
 34:   /** Constant for new instance types */
 35:   const TYPE_INSTANCE = 0x0010;
 36:   
 37:   /** Constant for shared instance types */
 38:   const TYPE_SHARED = 0x0100;
 39:   
 40:   /** Constant for aliases */
 41:   const TYPE_ALIAS = 0x1000;
 42:   
 43:   /** Singleton instance */
 44:   private static $_instance = null;
 45:   
 46:   /** The data container */
 47:   private $_store = array();
 48:   
 49:   /** The current endpoint in the data container */
 50:   private $_endPoint;
 51:   
 52:   /**
 53:    * Constructor should not be used.
 54:    * Use {@link getInstance()} instead.
 55:    */
 56:   public function __construct() { }
 57:   
 58:   /**
 59:    * Returns a singleton of the DependencyContainer.
 60:    * @return Swift_DependencyContainer
 61:    */
 62:   public static function getInstance()
 63:   {
 64:     if (!isset(self::$_instance))
 65:     {
 66:       self::$_instance = new self();
 67:     }
 68:     return self::$_instance;
 69:   }
 70:   
 71:   /**
 72:    * List the names of all items stored in the Container.
 73:    * @return array
 74:    */
 75:   public function listItems()
 76:   {
 77:     return array_keys($this->_store);
 78:   }
 79:   
 80:   /**
 81:    * Test if an item is registered in this container with the given name.
 82:    * @param string $itemName
 83:    * @return boolean
 84:    * @see register()
 85:    */
 86:   public function has($itemName)
 87:   {
 88:     return array_key_exists($itemName, $this->_store)
 89:       && isset($this->_store[$itemName]['lookupType']);
 90:   }
 91:   
 92:   /**
 93:    * Lookup the item with the given $itemName.
 94:    * @param string $itemName
 95:    * @return mixed
 96:    * @throws Swift_DependencyException If the dependency is not found
 97:    * @see register()
 98:    */
 99:   public function lookup($itemName)
100:   {
101:     if (!$this->has($itemName))
102:     {
103:       throw new Swift_DependencyException(
104:         'Cannot lookup dependency "' . $itemName . '" since it is not registered.'
105:         );
106:     }
107:     
108:     switch ($this->_store[$itemName]['lookupType'])
109:     {
110:       case self::TYPE_ALIAS:
111:         return $this->_createAlias($itemName);
112:       case self::TYPE_VALUE:
113:         return $this->_getValue($itemName);
114:       case self::TYPE_INSTANCE:
115:         return $this->_createNewInstance($itemName);
116:       case self::TYPE_SHARED:
117:         return $this->_createSharedInstance($itemName);
118:     }
119:   }
120:   
121:   /**
122:    * Create an array of arguments passed to the constructor of $itemName.
123:    * @param string $itemName
124:    * @return array
125:    */
126:   public function createDependenciesFor($itemName)
127:   {
128:     $args = array();
129:     if (isset($this->_store[$itemName]['args']))
130:     {
131:       $args = $this->_resolveArgs($this->_store[$itemName]['args']);
132:     }
133:     return $args;
134:   }
135:   
136:   /**
137:    * Register a new dependency with $itemName.
138:    * This method returns the current DependencyContainer instance because it
139:    * requires the use of the fluid interface to set the specific details for the
140:    * dependency.
141:    *
142:    * @param string $itemName
143:    * @return Swift_DependencyContainer
144:    * @see asNewInstanceOf(), asSharedInstanceOf(), asValue()
145:    */
146:   public function register($itemName)
147:   {
148:     $this->_store[$itemName] = array();
149:     $this->_endPoint =& $this->_store[$itemName];
150:     return $this;
151:   }
152:   
153:   /**
154:    * Specify the previously registered item as a literal value.
155:    * {@link register()} must be called before this will work.
156:    *
157:    * @param mixed $value
158:    * @return Swift_DependencyContainer
159:    */
160:   public function asValue($value)
161:   {
162:     $endPoint =& $this->_getEndPoint();
163:     $endPoint['lookupType'] = self::TYPE_VALUE;
164:     $endPoint['value'] = $value;
165:     return $this;
166:   }
167:   
168:   /**
169:    * Specify the previously registered item as an alias of another item.
170:    * @param string $lookup
171:    * @return Swift_DependencyContainer
172:    */
173:   public function asAliasOf($lookup)
174:   {
175:     $endPoint =& $this->_getEndPoint();
176:     $endPoint['lookupType'] = self::TYPE_ALIAS;
177:     $endPoint['ref'] = $lookup;
178:     return $this;
179:   }
180:   
181:   /**
182:    * Specify the previously registered item as a new instance of $className.
183:    * {@link register()} must be called before this will work.
184:    * Any arguments can be set with {@link withDependencies()},
185:    * {@link addConstructorValue()} or {@link addConstructorLookup()}.
186:    *
187:    * @param string $className
188:    * @return Swift_DependencyContainer
189:    * @see withDependencies(), addConstructorValue(), addConstructorLookup()
190:    */
191:   public function asNewInstanceOf($className)
192:   {
193:     $endPoint =& $this->_getEndPoint();
194:     $endPoint['lookupType'] = self::TYPE_INSTANCE;
195:     $endPoint['className'] = $className;
196:     return $this;
197:   }
198:   
199:   /**
200:    * Specify the previously registered item as a shared instance of $className.
201:    * {@link register()} must be called before this will work.
202:    * @param string $className
203:    * @return Swift_DependencyContainer
204:    */
205:   public function asSharedInstanceOf($className)
206:   {
207:     $endPoint =& $this->_getEndPoint();
208:     $endPoint['lookupType'] = self::TYPE_SHARED;
209:     $endPoint['className'] = $className;
210:     return $this;
211:   }
212:   
213:   /**
214:    * Specify a list of injected dependencies for the previously registered item.
215:    * This method takes an array of lookup names.
216:    * 
217:    * @param array $lookups
218:    * @return Swift_DependencyContainer
219:    * @see addConstructorValue(), addConstructorLookup()
220:    */
221:   public function withDependencies(array $lookups)
222:   {
223:     $endPoint =& $this->_getEndPoint();
224:     $endPoint['args'] = array();
225:     foreach ($lookups as $lookup)
226:     {
227:       $this->addConstructorLookup($lookup);
228:     }
229:     return $this;
230:   }
231:   
232:   /**
233:    * Specify a literal (non looked up) value for the constructor of the
234:    * previously registered item.
235:    * 
236:    * @param mixed $value
237:    * @return Swift_DependencyContainer
238:    * @see withDependencies(), addConstructorLookup()
239:    */
240:   public function addConstructorValue($value)
241:   {
242:     $endPoint =& $this->_getEndPoint();
243:     if (!isset($endPoint['args']))
244:     {
245:       $endPoint['args'] = array();
246:     }
247:     $endPoint['args'][] = array('type' => 'value', 'item' => $value);
248:     return $this;
249:   }
250:   
251:   /**
252:    * Specify a dependency lookup for the constructor of the previously
253:    * registered item.
254:    * 
255:    * @param string $lookup
256:    * @return Swift_DependencyContainer
257:    * @see withDependencies(), addConstructorValue()
258:    */
259:   public function addConstructorLookup($lookup)
260:   {
261:     $endPoint =& $this->_getEndPoint();
262:     if (!isset($this->_endPoint['args']))
263:     {
264:       $endPoint['args'] = array();
265:     }
266:     $endPoint['args'][] = array('type' => 'lookup', 'item' => $lookup);
267:     return $this;
268:   }
269:   
270:   // -- Private methods
271:   
272:   /** Get the literal value with $itemName */
273:   private function _getValue($itemName)
274:   {
275:     return $this->_store[$itemName]['value'];
276:   }
277:   
278:   /** Resolve an alias to another item */
279:   private function _createAlias($itemName)
280:   {
281:     return $this->lookup($this->_store[$itemName]['ref']);
282:   }
283:   
284:   /** Create a fresh instance of $itemName */
285:   private function _createNewInstance($itemName)
286:   {
287:     $reflector = new ReflectionClass($this->_store[$itemName]['className']);
288:     if ($reflector->getConstructor())
289:     {
290:       return $reflector->newInstanceArgs(
291:         $this->createDependenciesFor($itemName)
292:         );
293:     }
294:     else
295:     {
296:       return $reflector->newInstance();
297:     }
298:   }
299:   
300:   /** Create and register a shared instance of $itemName */
301:   private function _createSharedInstance($itemName)
302:   {
303:     if (!isset($this->_store[$itemName]['instance']))
304:     {
305:       $this->_store[$itemName]['instance'] = $this->_createNewInstance($itemName);
306:     }
307:     return $this->_store[$itemName]['instance'];
308:   }
309:   
310:   /** Get the current endpoint in the store */
311:   private function &_getEndPoint()
312:   {
313:     if (!isset($this->_endPoint))
314:     {
315:       throw new BadMethodCallException(
316:         'Component must first be registered by calling register()'
317:         );
318:     }
319:     return $this->_endPoint;
320:   }
321:   
322:   /** Get an argument list with dependencies resolved */
323:   private function _resolveArgs(array $args)
324:   {
325:     $resolved = array();
326:     foreach ($args as $argDefinition)
327:     {
328:       switch ($argDefinition['type'])
329:       {
330:         case 'lookup':
331:           $resolved[] = $this->_lookupRecursive($argDefinition['item']);
332:           break;
333:         case 'value':
334:           $resolved[] = $argDefinition['item'];
335:           break;
336:       }
337:     }
338:     return $resolved;
339:   }
340:   
341:   /** Resolve a single dependency with an collections */
342:   private function _lookupRecursive($item)
343:   {
344:     if (is_array($item))
345:     {
346:       $collection = array();
347:       foreach ($item as $k => $v)
348:       {
349:         $collection[$k] = $this->_lookupRecursive($v);
350:       }
351:       return $collection;
352:     }
353:     else
354:     {
355:       return $this->lookup($item);
356:     }
357:   }
358:   
359: }
