 26: /**
 27:  * An abstract base MIME Header.
 28:  * @package Swift
 29:  * @subpackage Mime
 30:  * @author Chris Corbyn
 31:  */
 32: class Swift_Mime_Headers_ParameterizedHeader
 33:   extends Swift_Mime_Headers_UnstructuredHeader
 34:   implements Swift_Mime_ParameterizedHeader
 35: {
 36:   
 37:   /**
 38:    * The Encoder used to encode the parameters.
 39:    * @var Swift_Encoder
 40:    * @access private
 41:    */
 42:   private $_paramEncoder;
 43:   
 44:   /**
 45:    * The parameters as an associative array.
 46:    * @var string[]
 47:    * @access private
 48:    */
 49:   private $_params = array();
 50:   
 51:   /**
 52:    * RFC 2231's definition of a token.
 53:    * @var string
 54:    * @access private
 55:    */
 56:   private $_tokenRe;
 57:   
 58:   /**
 59:    * Creates a new ParameterizedHeader with $name.
 60:    * @param string $name
 61:    * @param Swift_Mime_HeaderEncoder $encoder
 62:    * @param Swift_Encoder $paramEncoder, optional
 63:    */ 
 64:   public function __construct($name, Swift_Mime_HeaderEncoder $encoder,
 65:     Swift_Encoder $paramEncoder = null)
 66:   {
 67:     $this->setFieldName($name);
 68:     $this->setEncoder($encoder);
 69:     $this->_paramEncoder = $paramEncoder;
 70:     $this->initializeGrammar();
 71:     $this->_tokenRe = '(?:[\x21\x23-\x27\x2A\x2B\x2D\x2E\x30-\x39\x41-\x5A\x5E-\x7E]+)';
 72:   }
 73:   
 74:   /**
 75:    * Get the type of Header that this instance represents.
 76:    * @return int
 77:    * @see TYPE_TEXT, TYPE_PARAMETERIZED, TYPE_MAILBOX
 78:    * @see TYPE_DATE, TYPE_ID, TYPE_PATH
 79:    */
 80:   public function getFieldType()
 81:   {
 82:     return self::TYPE_PARAMETERIZED;
 83:   }
 84:   
 85:   /**
 86:    * Set the character set used in this Header.
 87:    * @param string $charset
 88:    */
 89:   public function setCharset($charset)
 90:   {
 91:     parent::setCharset($charset);
 92:     if (isset($this->_paramEncoder))
 93:     {
 94:       $this->_paramEncoder->charsetChanged($charset);
 95:     }
 96:   }
 97:   
 98:   /**
 99:    * Set the value of $parameter.
100:    * @param string $parameter
101:    * @param string $value
102:    */
103:   public function setParameter($parameter, $value)
104:   {
105:     $this->setParameters(array_merge($this->getParameters(), array($parameter => $value)));
106:   }
107:   
108:   /**
109:    * Get the value of $parameter.
110:    * @return string
111:    */
112:   public function getParameter($parameter)
113:   {
114:     $params = $this->getParameters();
115:     return array_key_exists($parameter, $params)
116:       ? $params[$parameter]
117:       : null;
118:   }
119:   
120:   /**
121:    * Set an associative array of parameter names mapped to values.
122:    * @param string[]
123:    */
124:   public function setParameters(array $parameters)
125:   {
126:     $this->clearCachedValueIf($this->_params != $parameters);
127:     $this->_params = $parameters;
128:   }
129:   
130:   /**
131:    * Returns an associative array of parameter names mapped to values.
132:    * @return string[]
133:    */
134:   public function getParameters()
135:   {
136:     return $this->_params;
137:   }
138:   
139:   /**
140:    * Get the value of this header prepared for rendering.
141:    * @return string
142:    */
143:   public function getFieldBody() //TODO: Check caching here
144:   {
145:     $body = parent::getFieldBody();
146:     foreach ($this->_params as $name => $value)
147:     {
148:       if (!is_null($value))
149:       {
150:         //Add the parameter
151:         $body .= '; ' . $this->_createParameter($name, $value);
152:       }
153:     }
154:     return $body;
155:   }
156:   
157:   // -- Protected methods
158:   
159:   /**
160:    * Generate a list of all tokens in the final header.
161:    * This doesn't need to be overridden in theory, but it is for implementation
162:    * reasons to prevent potential breakage of attributes.
163:    * @return string[]
164:    * @access protected
165:    */
166:   protected function toTokens($string = null)
167:   {
168:     $tokens = parent::toTokens(parent::getFieldBody());
169:     
170:     //Try creating any parameters
171:     foreach ($this->_params as $name => $value)
172:     {
173:       if (!is_null($value))
174:       {
175:         //Add the semi-colon separator
176:         $tokens[count($tokens)-1] .= ';';
177:         $tokens = array_merge($tokens, $this->generateTokenLines(
178:           ' ' . $this->_createParameter($name, $value)
179:           ));
180:       }
181:     }
182:     
183:     return $tokens;
184:   }
185:   
186:   // -- Private methods
187:   
188:   /**
189:    * Render a RFC 2047 compliant header parameter from the $name and $value.
190:    * @param string $name
191:    * @param string $value
192:    * @return string
193:    * @access private
194:    */
195:   private function _createParameter($name, $value)
196:   {
197:     $origValue = $value;
198:     
199:     $encoded = false;
200:     //Allow room for parameter name, indices, "=" and DQUOTEs
201:     $maxValueLength = $this->getMaxLineLength() - strlen($name . '=*N"";') - 1;
202:     $firstLineOffset = 0;
203:     
204:     //If it's not already a valid parameter value...
205:     if (!preg_match('/^' . $this->_tokenRe . '$/D', $value))
206:     {
207:       //TODO: text, or something else??
208:       //... and it's not ascii
209:       if (!preg_match('/^' . $this->getGrammar('text') . '*$/D', $value))
210:       {
211:         $encoded = true;
212:         //Allow space for the indices, charset and language
213:         $maxValueLength = $this->getMaxLineLength() - strlen($name . '*N*="";') - 1;
214:         $firstLineOffset = strlen(
215:           $this->getCharset() . "'" . $this->getLanguage() . "'"
216:           );
217:       }
218:     }
219:     
220:     //Encode if we need to
221:     if ($encoded || strlen($value) > $maxValueLength)
222:     {
223:       if (isset($this->_paramEncoder))
224:       {
225:         $value = $this->_paramEncoder->encodeString(
226:           $origValue, $firstLineOffset, $maxValueLength
227:           );
228:       }
229:       else //We have to go against RFC 2183/2231 in some areas for interoperability
230:       {
231:         $value = $this->getTokenAsEncodedWord($origValue);
232:         $encoded = false;
233:       }
234:     }
235:     
236:     $valueLines = isset($this->_paramEncoder) ? explode("\r\n", $value) : array($value);
237:     
238:     //Need to add indices
239:     if (count($valueLines) > 1)
240:     {
241:       $paramLines = array();
242:       foreach ($valueLines as $i => $line)
243:       {
244:         $paramLines[] = $name . '*' . $i .
245:           $this->_getEndOfParameterValue($line, $encoded, $i == 0);
246:       }
247:       return implode(";\r\n ", $paramLines);
248:     }
249:     else
250:     {
251:       return $name . $this->_getEndOfParameterValue(
252:         $valueLines[0], $encoded, true
253:         );
254:     }
255:   }
256:   
257:   /**
258:    * Returns the parameter value from the "=" and beyond.
259:    * @param string $value to append
260:    * @param boolean $encoded
261:    * @param boolean $firstLine
262:    * @return string
263:    * @access private
264:    */
265:   private function _getEndOfParameterValue($value, $encoded = false, $firstLine = false)
266:   {
267:     if (!preg_match('/^' . $this->_tokenRe . '$/D', $value))
268:     {
269:       $value = '"' . $value . '"';
270:     }
271:     $prepend = '=';
272:     if ($encoded)
273:     {
274:       $prepend = '*=';
275:       if ($firstLine)
276:       {
277:         $prepend = '*=' . $this->getCharset() . "'" . $this->getLanguage() .
278:           "'";
279:       }
280:     }
281:     return $prepend . $value;
282:   }

284: }
